(static) aRemove(arr, idxArr) → {Array}

Removes items from array by their indexes, just pass index itself if there is only one item needed to be removed. This function will not modify the original array.

var a = [1, 2 , 3];
S.aRemove(a, [0, 1]); // a => [3]

a = [1, 2 , 3];
sop.aRemove(a, 0);      // a => [2, 3]

 a = [1, 2 , 3];
sop.aRemove(a, [0]);    // a => [2, 3]

(static) getValueByPath(obj, path, sepopt) → {object|null}

Gets the value of obj by path:

var obj = {
     a : {
         b : {
             c : 1
         }
     }
}

var v = sop.getValueByPath(obj, 'a/b/c');
console.log(v); // v is 1
